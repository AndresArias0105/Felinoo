CREATE TABLE public.adoption_requests (
  id_request bigint GENERATED ALWAYS AS IDENTITY NOT NULL,
  id_user bigint NOT NULL,
  id_cat bigint NOT NULL,
  estado character varying NOT NULL DEFAULT '''pendiente'''::character varying,
  request_date timestamp with time zone NOT NULL DEFAULT now(),
  CONSTRAINT adoption_requests_pkey PRIMARY KEY (id_request),
  CONSTRAINT adoption_requests_id_user_fkey FOREIGN KEY (id_user) REFERENCES public.users(id_user),
  CONSTRAINT adoption_requests_id_cat_fkey FOREIGN KEY (id_cat) REFERENCES public.cats(id_cat)
);

CREATE TABLE public.cats (
  id_cat bigint GENERATED ALWAYS AS IDENTITY NOT NULL,
  name character varying NOT NULL,
  age integer NOT NULL,
  description character varying NOT NULL,
  img_url character varying NOT NULL,
  estado character varying NOT NULL,
  created_at timestamp with time zone NOT NULL DEFAULT now(),
  CONSTRAINT cats_pkey PRIMARY KEY (id_cat)
);

CREATE TABLE public.donations (
  id_donation bigint GENERATED ALWAYS AS IDENTITY NOT NULL,
  id_user bigint NOT NULL,
  amount double precision NOT NULL,
  payment_method character varying NOT NULL,
  donation_date timestamp with time zone NOT NULL DEFAULT now(),
  CONSTRAINT donations_pkey PRIMARY KEY (id_donation),
  CONSTRAINT donations_id_user_fkey FOREIGN KEY (id_user) REFERENCES public.users(id_user)
);

CREATE TABLE public.rehome_requests (
  id_request bigint GENERATED ALWAYS AS IDENTITY NOT NULL,
  id_user bigint,
  cat_name character varying NOT NULL,
  cat_age integer NOT NULL,
  cat_description character varying NOT NULL,
  img_url character varying NOT NULL,
  estado character varying NOT NULL,
  request_date timestamp with time zone NOT NULL DEFAULT now(),
  CONSTRAINT rehome_requests_pkey PRIMARY KEY (id_request),
  CONSTRAINT reubication_requests_id_user_fkey FOREIGN KEY (id_user) REFERENCES public.users(id_user)
);

CREATE TABLE public.users (
  id_user bigint GENERATED ALWAYS AS IDENTITY NOT NULL,
  fullname character varying NOT NULL,
  email character varying NOT NULL UNIQUE,
  phone character varying NOT NULL,
  username character varying NOT NULL UNIQUE,
  password_hash character varying NOT NULL,
  role character varying NOT NULL DEFAULT '''user'''::character varying,
  created_at timestamp without time zone DEFAULT (now() AT TIME ZONE 'utc'::text),
  CONSTRAINT users_pkey PRIMARY KEY (id_user)
);
